MySQL

Соединение — MySQL-пользователем root с паролем 1234.

mysql -u root -p
mysql -u root -p имя_базы 		войти в REPL-интерфейс MySQL (будет запрошен пароль MySQL-пользователя root)
позволяет давать SQL-запросы и видеть их результаты;
SQL-запросы ОБЯЗАТЕЛЬНО заканчивать точкой с запятой;
выход из REPL — команда quit или exit
nano /etc/mysql/mysql.conf.d/mysqld.cnf открыть редактор nano для изменения файла конфигурации MySQL;
выход из редактора — Ctrl+X
service mysql status
service mysql stop
service mysql start
service mysql restart 		посмотреть состояние сервера MySQL;
остановить сервер MySQL;
запустить сервер MySQL;
перезапустить сервер MySQL (например, если изменена конфигурация)
(для выполнения некоторых команд будет запрошен пароль Linux-пользователя user)

nginx

nano /etc/nginx/nginx.conf открыть редактор nano для изменения файла конфигурации nginx;
выход из редактора — Ctrl+X
sudo nginx -t проверить конфигурацию
(будет запрошен пароль Linux-пользователя user)
sudo nginx -s reload перезагрузить конфигурацию
(будет запрошен пароль Linux-пользователя user)
